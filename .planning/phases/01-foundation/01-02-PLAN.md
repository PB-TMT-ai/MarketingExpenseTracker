period_type = pgEnum("period_type", ["month","quarter","fy"])            // D-10

periods:        id bigserial PK, type period_type NOT NULL, label text NOT NULL,
                start_date date NOT NULL, end_date date NOT NULL,
                is_active boolean NOT NULL default false,                 // exactly one true, enforced app-side (D-11)
                created_at timestamptz NOT NULL defaultNow

plan_rows:      id bigserial PK,
                period_id bigint NOT NULL references periods.id,          // PRD-02 scoping
                activity text NOT NULL,                                   // registry discriminator
                sfid text NOT NULL,                                       // ALWAYS text (PITFALLS coercion)
                region/state/district/taluka/distributor/dealer text,     // shared who/where = real indexed cols
                planned_cost numeric(14,2),                               // budget (D-05) — numeric, never float
                fields jsonb NOT NULL default {}                          // plan-side activity extras
                CONSTRAINTS:
                  unique("plan_rows_match_key").on(period_id, activity, sfid)     // D-02 match key
                  index("plan_rows_filter_idx").on(period_id, activity, region, state, district)

executions:     id bigserial PK,
                plan_row_id bigint NOT NULL references planRows.id { onDelete: "restrict" },  // D-01/D-03 — THE GUARD
                  // NO sfid column here, by design (Anti-Pattern: an sfid on executions reopens the bypass)
                status text, unit_no text,                                // per-unit identity, Wall/Shop No (D-03)
                per_unit_cost numeric(14,2), total_cost numeric(14,2), total_sqft numeric(14,2),  // D-05 app-computed
                fields jsonb NOT NULL default {},                         // measurements / lat-long
                version integer NOT NULL default 0,                       // D-04 per-unit optimistic concurrency
                updated_at timestamptz NOT NULL defaultNow
                CONSTRAINTS: index("executions_plan_row_idx").on(plan_row_id)

execution_items: id bigserial PK,                                         // POP/Dealer-Kit children (D-07/D-08)
                execution_id bigint NOT NULL references executions.id { onDelete: "restrict" },
                item_name text NOT NULL,                                  // SNAPSHOT at entry (D-08), NOT an FK to item_master
                qty numeric(14,2) NOT NULL, rate numeric(14,2) NOT NULL,  // rate fresh per line (D-07)
                line_total numeric(14,2) NOT NULL                         // computed app-side

item_master:    id bigserial PK, name text NOT NULL, category text,      // optional grouping (D-06)
                active boolean NOT NULL default true                      // retire w/o hard delete (D-09)
</interfaces>

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| any write path → executions table | The only legitimate way to record spend; must be impossible to attach to a non-planned SFID |
| migration runner → live database | Schema changes apply to the local PGlite (and later Supabase) DB; must be the single generated source |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-01 | Tampering | off-plan write via a forgotten code path | mitigate | `executions.plan_row_id NOT NULL REFERENCES plan_rows(id) ON DELETE RESTRICT` with NO `sfid` column on executions — the DB rejects any orphan execution regardless of app code (COMP-01). |
| T-02-02 | Tampering | destructive plan re-upload erasing actuals | mitigate | `ON DELETE RESTRICT` (not CASCADE) from plan_rows → executions makes a destructive delete fail loudly; sets up Phase 2 PLAN-06 as a structural guarantee. |
| T-02-03 | Repudiation | ambiguous match key / silent duplicate plan rows | mitigate | Composite `UNIQUE (period_id, activity, sfid)` rejects duplicate plan rows at the DB; the importer (Phase 2) flags duplicates rather than guessing. |
| T-02-04 | Information Disclosure | money drift eroding the system-of-record | mitigate | All money/measure columns `numeric(14,2)`; NO float; NO jsonb→numeric generated columns (immutability/crash trap). Totals persisted as plain numeric, computed app-side (D-05). |
| T-02-05 | Tampering | schema drift between local and prod | mitigate | One `drizzle-kit generate` SQL source applied to PGlite via `migrate()` and to Supabase via `drizzle-kit migrate` — identical DDL both sides. |
| T-02-SC | Tampering | drizzle-kit / pglite install at migration time | mitigate | Already covered by RESEARCH Package Legitimacy Audit (Approved, no postinstall); no new packages introduced here. |
</threat_model>
